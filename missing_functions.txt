function renderCalendarView(todos) {
                const container = document.getElementById('calendarContainer');
                if (!container) return;

                const now = new Date();
                const year = now.getFullYear();
                const month = now.getMonth();
                const firstDay = new Date(year, month, 1);
                const lastDay = new Date(year, month + 1, 0);
                const daysInMonth = lastDay.getDate();
                const startingDayOfWeek = firstDay.getDay();

                container.innerHTML = '';

                // Day headers
                const dayNames = ['Sun', 'Mon', 'Tue', 'Wed', 'Thu', 'Fri', 'Sat'];
                dayNames.forEach(day => {
                    const header = document.createElement('div');
                    header.style.cssText = 'text-align: center; font-weight: 700; color: #64748b; padding: 12px; background: #f8fafc; border-radius: 8px;';
                    header.textContent = day;
                    container.appendChild(header);
                });

                // Empty cells for days before month starts
                for (let i = 0; i < startingDayOfWeek; i++) {
                    const empty = document.createElement('div');
                    empty.style.cssText = 'min-height: 100px; background: #f8fafc; border-radius: 8px;';
                    container.appendChild(empty);
                }

                // Days of month
                for (let day = 1; day <= daysInMonth; day++) {
                    const dayCell = document.createElement('div');
                    dayCell.style.cssText = 'min-height: 100px; padding: 8px; background: white; border-radius: 8px; border: 1px solid #e2e8f0;';

                    const dayHeader = document.createElement('div');
                    dayHeader.style.cssText = 'font-weight: 700; margin-bottom: 8px; color: #1E293B;';
                    dayHeader.textContent = day;
                    dayCell.appendChild(dayHeader);

                    const dayDate = new Date(year, month, day);
                    const dayTodos = todos.filter(t => {
                        if (!t.due_date) return false;
                        const due = new Date(t.due_date);
                        return due.toDateString() === dayDate.toDateString();
                    });

                    dayTodos.forEach(todo => {
                        const todoDot = document.createElement('div');
                        todoDot.style.cssText = 'padding: 4px 8px; margin: 4px 0; background: #e0e7ff; border-radius: 4px; font-size: 12px; cursor: pointer; overflow: hidden; text-overflow: ellipsis; white-space: nowrap;';
                        todoDot.textContent = todo.title;
                        todoDot.onclick = () => openTodoDetail(todo.id);
                        dayCell.appendChild(todoDot);
                    });

                    container.appendChild(dayCell);
                }
            }

function saveTodoDraftSnapshot() {
                const modal = document.getElementById('todoModal');
                if (!modal || modal.style.display !== 'flex') return;

                try {
                    const draft = collectTodoDraftData();
                    const key = getTodoDraftContextKey(draft.mode, draft.todoId);
                    const drafts = readTodoDraftCollection();
                    const nextDrafts = drafts.filter(d => getTodoDraftContextKey(d.mode, d.todoId) !== key);
                    nextDrafts.push(draft);
                    writeTodoDraftCollection(nextDrafts);
                } catch (error) {
                    console.warn('Could not save todo draft:', error);
                }
            }

function updateOurGoalPlaceholder() {
                const editor = document.getElementById('todoOurGoal');
                const placeholder = document.getElementById('todoOurGoalPlaceholder');
                if (!editor) return;

                // Get text content (strips all HTML tags)
                const text = editor.textContent.trim();
                const html = editor.innerHTML.trim();

                // Check if there's any actual content
                // Remove all HTML tags and check if there's any text
                const tempDiv = document.createElement('div');
                tempDiv.innerHTML = html;
                const plainText = tempDiv.textContent || tempDiv.innerText || '';
                const hasText = plainText.trim().length > 0;

                // Check for empty HTML states
                const isEmptyHTML = !html ||
                    html === '<br>' ||
                    html === '<div><br></div>' ||
                    html === '<p><br></p>' ||
                    html === '<div></div>' ||
                    html === '<p></p>';

                // Check for empty lists (no text in list items)
                const lists = editor.querySelectorAll('ul, ol');
                let hasListContent = false;
                if (lists.length > 0) {
                    lists.forEach(list => {
                        const listText = list.textContent.trim();
                        if (listText && listText.length > 0) {
                            hasListContent = true;
                        }
                    });
                }

                // Check for any list items with actual text content
                const allListItems = editor.querySelectorAll('li');
                let hasListItemText = false;
                allListItems.forEach(li => {
                    const liText = li.textContent.trim();
                    // Check if list item has actual content (not just whitespace, breaks, or bullet markers)
                    if (liText && liText.length > 0 && liText !== '•' && !liText.match(/^\d+\.?\s*$/)) {
                        hasListItemText = true;
                    }
                });

                // Editor has content if: there's text OR list items with text OR lists with content
                const hasAnyContent = hasText || hasListItemText || hasListContent;

                // Editor is empty only if no content at all
                const isEmpty = !hasAnyContent && isEmptyHTML;

                if (isEmpty) {
                    editor.setAttribute('data-empty', 'true');
                    if (placeholder) {
                        placeholder.style.display = 'block';
                        // Ensure placeholder is plain text with no list formatting
                        placeholder.textContent = 'Enter your goal here... You can format text and create lists.';
                        placeholder.style.listStyle = 'none';
                        placeholder.style.paddingLeft = '0';
                        placeholder.style.marginLeft = '0';
                    }
                } else {
                    editor.removeAttribute('data-empty');
                    if (placeholder) placeholder.style.display = 'none';
                }

                // Ensure proper padding for lists to prevent bullets from collapsing into border
                lists.forEach(list => {
                    list.style.paddingLeft = '24px';
                    list.style.marginLeft = '0';
                    list.style.marginTop = '6px';
                    list.style.marginBottom = '6px';
                });

                // Apply margin to list items (using allListItems already declared above)
                allListItems.forEach(li => {
                    li.style.marginBottom = '4px';
                });
            }

function uploadSingleCreationAttachment(file, isLast) {
                const token = authToken;
                if (!token) {
                    showToast('Error: Not authenticated', 'error');
                    return;
                }

                // Check file size (max 2GB)
                const maxSize = 2 * 1024 * 1024 * 1024; // 2GB
                if (file.size > maxSize) {
                    showToast(`"${file.name}" is too large. Maximum 2GB allowed.`, 'error');
                    return;
                }

                const uploadProgressId = createTaskCreationAttachmentUploadProgress(file);
                const uploadStartTime = Date.now();
                let simulatedPercent = 58;
                let simulatedTimer = null;

                const getEta = (progress) => {
                    const elapsedSeconds = Math.max(1, (Date.now() - uploadStartTime) / 1000);
                    if (progress <= 0) return Infinity;
                    const estimatedTotal = elapsedSeconds / (progress / 100);
                    return Math.max(0, estimatedTotal - elapsedSeconds);
                };

                const reader = new FileReader();

                reader.onprogress = function (event) {
                    if (!event.lengthComputable) return;
                    const readingPercent = Math.max(5, Math.min(55, Math.round((event.loaded / event.total) * 55)));
                    updateTaskCreationAttachmentUploadProgress(uploadProgressId, readingPercent, `Reading local file (${formatBytesForUpload(event.loaded)} / ${formatBytesForUpload(event.total)})...`, getEta(readingPercent));
                };

                reader.onload = function (e) {
                    try {
                        // Extract base64 data (remove data URL prefix)
                        const result = e.target.result;
                        if (!result || !result.includes(',')) {
                            throw new Error('Invalid file data');
                        }

                        const base64Data = result.split(',')[1];
                        const fullFileName = file.name;
                        const mimeType = file.type || 'application/octet-stream';

                        updateTaskCreationAttachmentUploadProgress(uploadProgressId, 58, 'Uploading to Drive...', getEta(58));
                        simulatedTimer = setInterval(() => {
                            simulatedPercent = Math.min(94, simulatedPercent + 3);
                            updateTaskCreationAttachmentUploadProgress(uploadProgressId, simulatedPercent, 'Uploading to Drive...', getEta(simulatedPercent));
                        }, 450);

                        // Upload to Drive (using root folder)
                        google.script.run
                            .withSuccessHandler(function (uploadResult) {
                                if (uploadResult && uploadResult.success && uploadResult.file) {
                                    if (simulatedTimer) {
                                        clearInterval(simulatedTimer);
                                        simulatedTimer = null;
                                    }
                                    updateTaskCreationAttachmentUploadProgress(uploadProgressId, 100, 'Ready to save ✅');

                                    // Store attachment data in pending array
                                    if (!window.pendingTaskCreationAttachments) {
                                        window.pendingTaskCreationAttachments = [];
                                    }
                                    window.pendingTaskCreationAttachments.push({
                                        file_name: fullFileName,
                                        file_url: uploadResult.file.url,
                                        file_type: mimeType,
                                        file_size: file.size
                                    });

                                    showToast(`✅ "${fullFileName}" uploaded to Drive (will be linked when task is saved)`, 'success');
                                    displayTaskCreationAttachments();

                                    finishTaskCreationAttachmentUploadProgress(uploadProgressId, true, 'Ready ✅');
                                } else {
                                    if (simulatedTimer) {
                                        clearInterval(simulatedTimer);
                                        simulatedTimer = null;
                                    }
                                    finishTaskCreationAttachmentUploadProgress(uploadProgressId, false, 'Upload failed');
                                    const errorMsg = uploadResult?.error || 'Unknown error';
                                    console.error('Drive upload failed:', errorMsg);
                                    if (errorMsg.includes('Permission denied') || errorMsg.includes('Editor access')) {
                                        showToast(`❌ Permission error: Please refresh the page and try again.`, 'error');
                                    } else {
                                        showToast(`❌ Failed to upload "${fullFileName}": ${errorMsg}`, 'error');
                                    }
                                }
                            })
                            .withFailureHandler(function (error) {
                                if (simulatedTimer) {
                                    clearInterval(simulatedTimer);
                                    simulatedTimer = null;
                                }
                                finishTaskCreationAttachmentUploadProgress(uploadProgressId, false, 'Upload failed');
                                console.error('Upload error:', error);
                                const errorMsg = error.message || 'Upload failed';
                                if (errorMsg.includes('Permission denied') || errorMsg.includes('Editor access')) {
                                    showToast(`❌ Permission error: Please refresh the page and try again.`, 'error');
                                } else {
                                    showToast(`❌ Failed to upload "${fullFileName}": ${errorMsg}`, 'error');
                                }
                            })
                            .uploadFileToDrive(base64Data, fullFileName, mimeType, 'root', token, 'task_creation_pending');
                    } catch (err) {
                        if (simulatedTimer) {
                            clearInterval(simulatedTimer);
                            simulatedTimer = null;
                        }
                        finishTaskCreationAttachmentUploadProgress(uploadProgressId, false, 'File processing failed');
                        console.error('Error processing file:', err);
                        showToast(`❌ Error processing "${file.name}": ${err.message}`, 'error');
                    }
                };

                reader.onerror = function () {
                    finishTaskCreationAttachmentUploadProgress(uploadProgressId, false, 'File read failed');
                    showToast(`❌ Error reading file "${file.name}". Please try again.`, 'error');
                };

                reader.onabort = function () {
                    finishTaskCreationAttachmentUploadProgress(uploadProgressId, false, 'Upload cancelled');
                    showToast(`Upload cancelled for "${file.name}"`, 'info');
                };

                // Read file as base64
                reader.readAsDataURL(file);
            }
